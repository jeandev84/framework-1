<?php
namespace Jan\Component\Database;


use Jan\Component\Database\Exception\DatabaseException;

/**
 * Class Capsule
 * @package Jan\Component\Database
*/
class Capsule
{

    /**
     * @var DatabaseManager
    */
    protected static $instance;


    /**
     * @var array
    */
    protected $configParams = [];


    /**
     * add connection params
     *
     * @param array $configParams
     * @return Capsule
    */
    public function addConnection(array $configParams): Capsule
    {
        $this->configParams = $configParams;

        return $this;
    }



    /**
     * set globally connection to database
    */
    public function setAsGlobal()
    {
        if (! static::$instance) {
            $this->bootInstance();
        }
    }



    /**
     * @return DatabaseManager
     * @throws DatabaseException
    */
    public static function getInstance(): DatabaseManager
    {
        if (! static::$instance) {
            throw new DatabaseException('Cannot get instance of database manager.');
        }

        return static::$instance;
    }



    /**
     * @throws DatabaseException
    */
    public function bootInstance()
    {
        if (empty($this->configParams)) {
            throw new DatabaseException('You need to configure you database.');
        }

        static::$instance = new DatabaseManager($this->configParams);
    }
}
