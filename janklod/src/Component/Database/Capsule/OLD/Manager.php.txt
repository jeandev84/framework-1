<?php
namespace Jan\Component\Database\Capsule;


use Jan\Component\Database\DatabaseManager;


/**
 * Class Manager
 * @package Jan\Component\Database\Capsule
*/
class Manager
{

     /**
      * @var DatabaseManager
     */
     protected static $instance;



     protected $params  = [];




     /**
      * Manager constructor.
     */
     public function __construct()
     {
          if (! static::$instance) {
              static::$instance = new DatabaseManager();
          }
     }



     /**
       * @param string $name
       * @param array $configParams
      * @return Manager
     */
     public function addConnection(string $name, array $config): Manager
     {
         $this->params = [$name, $config];
         return $this;
     }




     /**
      * boot connection
     */
     public function bootAsGlobal(): Manager
     {
         list($name, $config) = $this->params;

         static::$instance->connectTo($name, $config);

         return $this;
     }



//     /**
//      * @param string|null $name
//      * @return mixed
//     */
//     public function connection(string $name = null)
//     {
//          return static::$instance->connection($name);
//     }




     /**
      * @return mixed
      * @throws \Exception
     */
     public static function instance()
     {
         if (! static::$instance) {
             throw new \Exception('Cannot get globally instance of capsule.');
         }

         return static::$instance;
     }
}
